% Requires: tcolorbox, pgffor
\newcommand{\boxes}[1]{
    \noindent
    \begin{tcbraster}[raster force size=false, raster equal height]%
        \foreach \hobby in #1 {%
            \begin{tcolorbox}[fit, hbox]%
                \hobby%
            \end{tcolorbox}%
        }%
    \end{tcbraster}%
}
